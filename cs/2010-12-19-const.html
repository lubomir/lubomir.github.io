Jak se projevuje 'const'

zpět na hlavní stránku
19. prosinec 2010
Označeno jako: C, programování, const.

Toto je testovací kód, který se pochopitelně nezkompiluje, a i kdyby, nic rozumného dělat nebude. Stejně jako strcpy se chová i free.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main (void)
{
  const char *       str1 = (char*) malloc(10*sizeof(char));
  char * const       str2 = (char*) malloc(10*sizeof(char));
  const char * const str3 = (char*) malloc(10*sizeof(char));

  str1 = NULL;
  str2 = NULL;
  str3 = NULL;

  strcpy(str1, "a");
  strcpy(str2, "a");
  strcpy(str3, "a");

  return 0;
}
